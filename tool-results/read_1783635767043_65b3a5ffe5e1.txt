    52→The component has TWO drag systems. Only the zone-based one is rendered:

   100→### CHECK 2 — Touch conflicts on mobile (pan vs drag) — ❌ BUG
   101→
   102→The entire drag system relies on HTML5 drag-and-drop, which does NOT fire on touch devices. There is no pointer/touch fallback, no `touch-action` styling, and Framer Motion's `drag` is not used.
   103→
   104→Verbatim — TablesSection.tsx:698-705 (ZoneTable):
   105→```tsx
   106→<motion.div
   107→  initial={reduceMotion ? {} : { opacity: 0, scale: 0.8 }}
   108→  animate={{ opacity: 1, scale: 1 }}
   109→  transition={{ duration: 0.2 }}
   110→  draggable={editMode}
   111→  onDragStart={(e: any) => { if (editMode && onDragStart) onDragStart(e, table.id); }}
   112→  onDragEnd={() => { if (editMode && onDragEnd) onDragEnd(); }}
   113→  onMouseEnter={() => { onHover(table.id); setShowPopover(true); }}
   114→  onMouseLeave={() => { onHover(null); setShowPopover(false); }}
   115→  onClick={() => onSelect(table)}
   116→  whileHover={editMode ? {} : { scale: 1.1, zIndex: 10 }}
   117→  whileTap={{ scale: 0.95 }}
   118→```
   119→
   120→Canvas container (scroll-panning surface):
   121→```tsx
   122→// TablesSection.tsx:356
   123→<div className="space-y-3 max-h-[520px] overflow-y-auto rounded-xl pr-1" style={{ scrollbarWidth: "thin" }}>
   124→```
   125→
   126→Grep results across the file:
   127→- `touch-action` / `touchAction` — 0 hits.
   128→- `touch-none` Tailwind class — 0 hits.
   129→- `dragMomentum` — 0 hits (Framer Motion `drag` prop not used at all).
   130→- `dragConstraints` — 0 hits.
   131→- `onTouchStart` / `onPointerDown` / `onPointerMove` / `onPointerUp` — 0 hits.
   132→
   133→Consequences:
   134→- On iOS Safari and Android Chrome, `draggable={true}` does nothing — admins literally cannot rearrange tables from a phone/tablet.
   135→- The canvas scroll surface (`overflow-y-auto`) keeps `touch-action: auto`, so touch panning works, but since drag never fires there is no conflict — the feature simply does not work on touch.
   136→- Click-to-select still works (onClick fires on tap), but `onMouseEnter`/`onMouseLeave` (line 706-707) do not fire on touch, so the CRM popover (lines 747-776) is also effectively desktop-only.
   137→
   138→Proposed fix (recommended approach — Framer Motion pointer-based drag, which works for both mouse and touch):
   139→```tsx
   140→<motion.div
   141→  drag={editMode}
   142→  dragMomentum={false}
   143→  dragConstraints={containerRef}
   144→  dragElastic={0}
   145→  onDragStart={() => setDraggingId(table.id)}
   146→  onDragEnd={(e, info) => {
   147→    // snap-to-zone detection: find zone panel under pointer
   148→    handleDropByPointer(table.id, info.point);
   149→    setDraggingId(null);
   150→  }}
   151→  style={{ touchAction: editMode ? "none" : "auto" }} // ← critical for mobile
   152→  whileDrag={{ scale: 1.1, zIndex: 50 }}
   153→>
   154→```
   155→Or, minimally, add `style={{ touchAction: "none" }}` and `onTouchStart`/`onTouchMove`/`onTouchEnd` handlers that mimic `handleDragStart/Move/End`.

   217→### CHECK 4 — Z-index of selected table popover — ❌ BUG
   218→
   219→The popover is a child of the table's motion.div. The parent's z-index is raised to 10 only via `whileHover` (transient, only during hover), and the popover uses `z-30` — but that `z-30` is INSIDE the parent's stacking context, so a sibling table with the same z-index 10 (or any table later in DOM order while hover transitions) can render on top of the popover.
   220→
   221→Verbatim — ZoneTable parent motion.div (TablesSection.tsx:698-721):
   222→```tsx
   223→<motion.div
   224→  ...
   225→  whileHover={editMode ? {} : { scale: 1.1, zIndex: 10 }}    // ← zIndex 10 only while hovering
   226→  whileTap={{ scale: 0.95 }}
   227→  className={cn(
   228→    "border-2 flex flex-col items-center justify-center transition-all select-none cursor-pointer relative",
   229→    ...
   230→    (isHovered || isSelected) && "ring-2 ring-yellow-400 ring-offset-2 ring-offset-[#0a0a0a]", // ← NO z-index when selected
   231→    ...
   232→  )}
   233→>
   234→```
   235→
   236→Popover (TablesSection.tsx:746-756):
   237→```tsx
   238→<AnimatePresence>
   239→  {showPopover && !editMode && (
   240→    <motion.div
   241→      ...
   242→      className="absolute z-30 pointer-events-none"           // ← z-30 trapped inside parent z-10
   243→      style={{ bottom: "100%", left: "50%", transform: "translateX(-50%)", marginBottom: "4px" }}
   244→    >
   245→```
   246→
   247→Concrete failure modes:
   248→1. Two adjacent tables in the same flex row. User hovers table A (zIndex 10, popover shown at z-30 inside A's stacking context). User moves cursor toward the popover; mid-transition the hover state on A is still active AND pointer enters table B (B also gets zIndex 10). Because B comes later in DOM and has the same z-index, B paints OVER A's popover — popover visually clipped/hidden behind B.
   249→2. The popover is only triggered by `onMouseEnter` (line 706) — when a table is CLICKED (`isSelected`), no popover is shown at all; the user only gets the full Dialog. That part is fine. BUT the selected table itself is NOT elevated above its siblings, so its `ring-yellow-400` selection ring can be visually overlapped by neighbours.
   250→
   251→InteractiveTable (the dead-code sibling at lines 553-588) has the same pattern but is even more explicit about the bug:
   252→```tsx
   253→// TablesSection.tsx:558
   254→style={{ ..., zIndex: isDragging ? 20 : 1 }}   // ← only 20 while dragging, else 1
   255→// TablesSection.tsx:563
   256→whileHover={editMode ? {} : { scale: 1.12, zIndex: 10 }}
   257→// TablesSection.tsx:618
   258→className="absolute z-30 pointer-events-none"
   259→```
   260→
   261→Proposed fix — promote the parent's z-index when hovered OR selected via a stable Tailwind class (not a transient inline style), and bump the popover higher than any sibling can reach:
   262→```tsx
   263→<motion.div
   264→  ...
   265→  className={cn(
   266→    "... relative",
   267→    (isHovered || isSelected) && "z-50",                       // ← stable z-index elevation
   268→  )}
   269→  whileHover={editMode ? {} : { scale: 1.1 }}
   270→  whileTap={{ scale: 0.95 }}
   271→>
   272→  ...
   273→  <motion.div
   274→    className="absolute z-[60] pointer-events-none"            // ← above any sibling at z-50
   275→    style={{ bottom: "100%", left: "50%", transform: "translateX(-50%)", marginBottom: "4px" }}
   276→  />
   277→```

   281→### Other related issues found
   282→
   283→1. **Dead code (~170 lines).** `InteractiveTable` (lines 537-671), `pendingPositions` state (line 104), `draggingId` state (line 105), `dragOffset` ref (line 108), `handleDragStart`/`handleDragMove`/`handleDragEnd` (lines 192-211), `savePositions` (lines 213-216), `getTablePos` (lines 266-270), `savePositionsMut` (lines 135-140), and the "Guardar" button (lines 287-289) are all unreachable. The orphan `/api/tables/positions` endpoint matches this dead UI.
   284→
   285→2. **No in-flight guard on zone-drop.** `changeZoneMut.mutate` is fired on every `drop` event without checking `changeZoneMut.isPending`. A user who drags table 1 to TERRACE then immediately drags table 2 to BAR triggers two parallel PATCHes; TanStack Query invalidates `["tables"]` twice, causing a double refetch and a possible visual flicker. Add `draggable={editMode && !changeZoneMut.isPending}` (or use `mutateAsync` with a queue).
   286→
   287→3. **No optimistic update on zone change.** After a successful drop the table visually snaps back to its original zone until the `["tables"]` refetch resolves (~150-400 ms). On slow networks this looks broken. Use TanStack Query's `onMutate` to write the new zone into the cache immediately and roll back on `onError`.
   288→
   289→4. **No reservation-side sync on zone drop.** `changeZoneMut` PATCHes only the table's `zone`. If a seated reservation (`status: SEATED`) is bound to that table, the reservation's `zone` field is NOT updated until the next time `/api/tables` is hit (which fetches reservations separately). The `transfer` route DOES sync reservation.zone (src/app/api/tables/transfer/route.ts:50-58) — zone drop does not.
   290→
   291→5. **Infinite animation cost.** Every occupied/reserved/preparing table spawns a `motion.span` with `animate={{ opacity: [0.4, 1, 0.4] }} transition={{ duration: 2, repeat: Infinity }}` (lines 722-730). For a restaurant with 40+ active tables this is 40 concurrent rAF loops. `useReducedMotion` (line 722) gates it for accessibility, but on a mid-tier Android tablet the LED-style boxShadow pulsing on `InteractiveTable` (dead) and the opacity pulse on `ZoneTable` (live) will burn CPU. Consider gating with `IntersectionObserver` so off-screen tables don't animate, or animate a single shared CSS keyframe.
   292→
   293→6. **Sequential UPDATEs in `/api/tables/positions`.** src/app/api/tables/positions/route.ts:21-36 loops with `for...of` + `await`. If 20 tables are batched, that's 20 sequential network round-trips. Use a single `supabaseAdmin.from('tables').update(...).in('id', ids)` (with a CASE WHEN for per-row positions) or a Postgres function.
   294→
   295→7. **Popover `pointer-events-none`.** TablesSection.tsx:754 (and 618 for the dead variant) sets `pointer-events-none` on the popover. This is intentional (the popover is a hover preview, click goes to the table), but it also means the popover can't contain a quick-action button. If product wants a "Traspasar" CTA in the preview, this needs revisiting.
   296→
   297→8. **No drag preview customisation.** HTML5 dnd uses the browser's default ghost (which includes the entire table card with `opacity-40` applied via `isDragging` class, line 718). On Chromium the ghost is a screenshot, on Firefox it can lag. A custom `dragImage` via `e.dataTransfer.setDragImage` would give a cleaner UX.
   298→
   299→9. **`onDragStart` typed as `any`** (line 704) — bypasses TS safety. Minor, but worth tightening to `React.DragEvent<HTMLDivElement>`.
   300→
   301→10. **No `aria-grabbed` / `role="application"`** on draggable tables — screen-reader users have no way to know tables are draggable. Pair with fix for CHECK 2 (touch/pointer drag) to also add keyboard handlers (Space to grab, arrows to move, Space to drop).

   829→### C4 — ⚠️ WARN · A10 SSRF — image proxy / Next.js image optimizer + web-import
   830→File: next.config.ts:9-15
   831→```ts
   832→images: {
   833→  remotePatterns: [
   834→    { protocol: "https", hostname: "**" },
   835→    { protocol: "http", hostname: "**" },
   836→  ],

   843→The Next.js image optimizer accepts ANY http(s) URL via `_next/image?url=…` and fetches it server-side. Still, allowing arbitrary SVG is risky if the CSP is ever loosened.

   848→1. Restrict `remotePatterns` to a whitelist of known image CDNs (Cloudinary, Cloudflare Images, Supabase Storage, the app's own origin) — drop `hostname: "**"`.

   850→3. In `isPrivateUrl`, also block `::ffff:`-prefixed IPv4, hostnames that resolve to private IPs (use `dns.lookup`), and well-known cloud metadata hostnames (`metadata.google.internal`, `metadata.azure.com`, `169.254.169.254`).

  1409→❌ BUG · 0015:6-53 · transfer_reservation RPC is SECURITY DEFINER with NO org check
  1410→```
  1411→CREATE OR REPLACE FUNCTION transfer_reservation(
  1412→  p_reservation_id UUID,
  1413→  p_old_table_id UUID,
  1414→  p_new_table_id UUID
  1415→)
  1416→RETURNS JSONB AS $$
  1417→...
  1418→  SELECT * INTO v_reservation FROM reservations WHERE id = p_reservation_id;
  1419→  ...
  1420→  UPDATE reservations SET table_id = p_new_table_id ... WHERE id = p_reservation_id;
  1421→  UPDATE tables SET status = 'AVAILABLE' ... WHERE id = p_old_table_id;
  1422→  UPDATE tables SET status = 'RESERVED' ... WHERE id = p_new_table_id;
  1423→...
  1424→$$ LANGUAGE plpgsql SECURITY DEFINER;
  1425→```
  1426→SECURITY DEFINER means the function runs as its owner (postgres, which bypasses RLS). Any authenticated user can call `select transfer_reservation(some_uuid, table_a, table_b);` and move ANY reservation in ANY tenant's table. There is no `where organization_id = current_user_org_id()` check anywhere.
  1427→Also: p_old_table_id is never validated to equal v_reservation.table_id, so a malicious caller could free an unrelated table by passing its UUID as p_old_table_id.
  1428→Fix:
  1429→```sql
  1430→create or replace function transfer_reservation(
  1431→  p_reservation_id uuid, p_old_table_id uuid, p_new_table_id uuid
  1432→) returns jsonb as $$
  1433→declare
  1434→  v_reservation record;
  1435→  v_new_table    record;
  1436→  v_org          uuid := current_user_org_id();
  1437→begin
  1438→  if v_org is null and not is_current_user_super_admin() then
  1439→    return jsonb_build_object('ok', false, 'error', 'No autenticado');
  1440→  end if;
  1441→
  1442→  select * into v_reservation from reservations
  1443→   where id = p_reservation_id
  1444→     and (v_org is null or organization_id = v_org);  -- super_admin bypasses
  1445→  if not found then
  1446→    return jsonb_build_object('ok', false, 'error', 'Reserva no encontrada');
  1447→  end if;
  1448→
  1449→  if v_reservation.table_id is distinct from p_old_table_id then
  1450→    return jsonb_build_object('ok', false, 'error', 'La mesa origen no coincide con la reserva');
  1451→  end if;
  1452→
  1453→  select * into v_new_table from tables
  1454→   where id = p_new_table_id
  1455→     and (v_org is null or organization_id = v_org);
  1456→  if not found then
  1457→    return jsonb_build_object('ok', false, 'error', 'Mesa de destino no encontrada');
  1458→  end if;
  1459→
  1460→  update reservations set table_id = p_new_table_id, zone = v_new_table.zone, updated_at = now()
  1461→   where id = p_reservation_id;
  1462→  if p_old_table_id is not null then
  1463→    update tables set status = 'AVAILABLE', updated_at = now() where id = p_old_table_id;
  1464→  end if;
  1465→  update tables set status = 'RESERVED', updated_at = now() where id = p_new_table_id;
  1466→
  1467→  return jsonb_build_object('ok', true, 'message', 'OK', 'new_table', row_to_json(v_new_table));
  1468→exception when others then
  1469→  return jsonb_build_object('ok', false, 'error', SQLERRM);
  1470→end;
  1471→$$ language plpgsql security definer set search_path = public, pg_temp;
  1472→```
  1473→(The `SET search_path` is critical for SECURITY DEFINER functions to prevent search_path hijacking.)
