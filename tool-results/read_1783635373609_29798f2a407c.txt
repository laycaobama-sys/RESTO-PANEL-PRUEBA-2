    92вҶ’-- 2. REESCRIBIR transfer_reservation() CON VALIDACIГ“N DE ORG
    93вҶ’-- ============================================================
    94вҶ’-- Antes: cualquier usuario podГӯa mover una reserva de cualquier tenant.
    95вҶ’-- Ahora: valida que la reserva, mesa origen y mesa destino pertenezcan
    96вҶ’-- a la organizaciГіn del JWT actual.
    97вҶ’--
    98вҶ’-- NOTA: PostgreSQL NO permite cambiar nombres de parГЎmetros con
    99вҶ’-- CREATE OR REPLACE FUNCTION. Hay que hacer DROP primero.
   100вҶ’-- Usamos la signatura completa (uuid, uuid, uuid) para no afectar
   101вҶ’-- a otras posibles versiones de la funciГіn.
   102вҶ’
   103вҶ’DROP FUNCTION IF EXISTS transfer_reservation(uuid, uuid, uuid);
   104вҶ’DROP FUNCTION IF EXISTS transfer_reservation(uuid, uuid);
   105вҶ’
   106вҶ’CREATE OR REPLACE FUNCTION transfer_reservation(
   107вҶ’  p_reservation_id uuid,
   108вҶ’  p_new_table_id uuid,
   109вҶ’  p_old_table_id uuid DEFAULT NULL
   110вҶ’)
   111вҶ’RETURNS boolean
   112вҶ’LANGUAGE plpgsql
   113вҶ’SECURITY DEFINER
   114вҶ’SET search_path = public, pg_temp
   115вҶ’AS $$
   116вҶ’DECLARE
   117вҶ’  v_reservation record;
   118вҶ’  v_new_table record;
   119вҶ’  v_org_id uuid;
   120вҶ’BEGIN
   121вҶ’  -- Get the caller's organization from the JWT claim.
   122вҶ’  v_org_id := current_user_org_id();
   123вҶ’  IF v_org_id IS NULL AND NOT is_current_user_super_admin() THEN
   124вҶ’    RAISE EXCEPTION 'No organization context';
   125вҶ’  END IF;
   126вҶ’
   127вҶ’  -- Load the reservation (must exist).
   128вҶ’  SELECT * INTO v_reservation
   129вҶ’  FROM reservations
   130вҶ’  WHERE id = p_reservation_id
   131вҶ’  FOR UPDATE;
   132вҶ’
   133вҶ’  IF NOT FOUND THEN
   134вҶ’    RAISE EXCEPTION 'Reservation not found';
   135вҶ’  END IF;
   136вҶ’
   137вҶ’  -- Org isolation: the reservation must belong to the caller's org
   138вҶ’  -- (unless super admin).
   139вҶ’  IF v_org_id IS NOT NULL AND v_reservation.organization_id != v_org_id THEN
   140вҶ’    RAISE EXCEPTION 'Forbidden: reservation does not belong to your organization';
   141вҶ’  END IF;
   142вҶ’
   143вҶ’  -- If p_old_table_id was provided, it must match the reservation's
   144вҶ’  -- current table_id (prevents race conditions where two clients
   145вҶ’  -- transfer the same reservation simultaneously).
   146вҶ’  IF p_old_table_id IS NOT NULL AND v_reservation.table_id IS NOT NULL
   147вҶ’     AND p_old_table_id::text != v_reservation.table_id::text THEN
   148вҶ’    RAISE EXCEPTION 'Old table id does not match reservation''s current table';
   149вҶ’  END IF;
   150вҶ’
   151вҶ’  -- Load the new table (must exist and belong to the same org).
   152вҶ’  SELECT * INTO v_new_table
   153вҶ’  FROM tables
   154вҶ’  WHERE id = p_new_table_id
   155вҶ’  FOR UPDATE;
   156вҶ’
   157вҶ’  IF NOT FOUND THEN
   158вҶ’    RAISE EXCEPTION 'Target table not found';
   159вҶ’  END IF;
   160вҶ’
   161вҶ’  IF v_org_id IS NOT NULL AND v_new_table.organization_id != v_org_id THEN
   162вҶ’    RAISE EXCEPTION 'Forbidden: target table does not belong to your organization';
   163вҶ’  END IF;
   164вҶ’
   165вҶ’  -- All three operations in a single transaction (atomic):
   166вҶ’  -- 1. Update reservation to point to new table.
   167вҶ’  -- 2. Free the old table (mark AVAILABLE) вҖ” only if it had this reservation.
   168вҶ’  -- 3. Mark the new table as RESERVED.
   169вҶ’  UPDATE reservations
   170вҶ’  SET table_id = p_new_table_id, updated_at = now()
   171вҶ’  WHERE id = p_reservation_id;
   172вҶ’
   173вҶ’  IF v_reservation.table_id IS NOT NULL THEN
   174вҶ’    UPDATE tables
   175вҶ’    SET status = 'AVAILABLE', updated_at = now()
   176вҶ’    WHERE id = v_reservation.table_id
   177вҶ’      AND organization_id = v_reservation.organization_id;
   178вҶ’  END IF;
   179вҶ’
   180вҶ’  UPDATE tables
   181вҶ’  SET status = 'RESERVED', updated_at = now()
   182вҶ’  WHERE id = p_new_table_id
   183вҶ’    AND organization_id = v_reservation.organization_id;
   184вҶ’
   185вҶ’  RETURN true;
   186вҶ’END;
   187вҶ’$$;
   188вҶ’
   189вҶ’COMMENT ON FUNCTION transfer_reservation IS 'Atomic transfer of a reservation between tables. Validates org ownership. All 3 operations (update reservation, free old table, reserve new table) happen in a single transaction with automatic rollback on failure.';

   206вҶ’-- 4. update_customer_metrics() con decremento al revertir
   207вҶ’-- ============================================================
   208вҶ’-- VersiГіn corregida: si una reserva pasa de COMPLETED a CANCELLED,
   209вҶ’-- resta 1 del contador visits_count (en lugar de solo sumar).
   210вҶ’--
   211вҶ’-- NOTA: Mismo problema que con transfer_reservation вҖ” DROP IF EXISTS
   212вҶ’-- primero para evitar el error 42P13 si la funciГіn existГӯa con otra
   213вҶ’-- signatura. CASCADE por si habГӯa triggers asociados.
   214вҶ’
   215вҶ’DROP FUNCTION IF EXISTS update_customer_metrics() CASCADE;
   216вҶ’
   217вҶ’CREATE OR REPLACE FUNCTION update_customer_metrics()
   218вҶ’RETURNS trigger
   219вҶ’LANGUAGE plpgsql
   220вҶ’SECURITY DEFINER
   221вҶ’SET search_path = public, pg_temp
   222вҶ’AS $$
   223вҶ’DECLARE
   224вҶ’  v_old_status text;
   225вҶ’  v_new_status text;
   226вҶ’  v_customer_id uuid;
   227вҶ’BEGIN
   228вҶ’  v_old_status := COALESCE(OLD.status, '');
   229вҶ’  v_new_status := COALESCE(NEW.status, '');
   230вҶ’
   231вҶ’  -- Only proceed if status changed.
   232вҶ’  IF v_old_status = v_new_status THEN
   233вҶ’    RETURN NEW;
   234вҶ’  END IF;
   235вҶ’
   236вҶ’  -- Find the customer (reservations store customer_id if linked).
   237вҶ’  v_customer_id := NEW.customer_id;
   238вҶ’  IF v_customer_id IS NULL THEN
   239вҶ’    RETURN NEW;
   240вҶ’  END IF;
   241вҶ’
   242вҶ’  -- Increment when entering COMPLETED.
   243вҶ’  IF v_new_status = 'COMPLETED' AND v_old_status != 'COMPLETED' THEN
   244вҶ’    UPDATE customers
   245вҶ’    SET visits_count = COALESCE(visits_count, 0) + 1,
   246вҶ’        last_visit_at = now(),
   247вҶ’        updated_at = now()
   248вҶ’    WHERE id = v_customer_id;
   249вҶ’  END IF;
   250вҶ’
   251вҶ’  -- Decrement when leaving COMPLETED (e.g., reversal/correction).
   252вҶ’  IF v_old_status = 'COMPLETED' AND v_new_status != 'COMPLETED' THEN
   253вҶ’    UPDATE customers
   254вҶ’    SET visits_count = GREATEST(0, COALESCE(visits_count, 1) - 1),
   255вҶ’        updated_at = now()
   256вҶ’    WHERE id = v_customer_id;
   257вҶ’  END IF;
   258вҶ’
   259вҶ’  RETURN NEW;
   260вҶ’END;
   261вҶ’$$;

   319вҶ’-- 6. UNIQUE EN subscription_history PARA IDEMPOTENCIA WEBHOOK
   320вҶ’-- ============================================================
   321вҶ’-- Stripe reenvГӯa eventos cuando recibe 500. Necesitamos un UNIQUE
   322вҶ’-- compuesto para que el ON CONFLICT del webhook funcione.
   323вҶ’-- Antes de crear el Гӯndice, eliminamos duplicados existentes.
   324вҶ’-- Envuelto en DO $$ para no fallar si subscription_history no existe.
   325вҶ’
   326вҶ’DO $$
   327вҶ’BEGIN
   328вҶ’  IF EXISTS (SELECT 1 FROM information_schema.tables
   329вҶ’             WHERE table_schema = 'public' AND table_name = 'subscription_history') THEN
   330вҶ’
   331вҶ’    -- 6a. De-duplicar subscription_history por (organization_id, event_type, details)
   332вҶ’    -- Mantener solo la fila mГЎs reciente (la de id mayor вҖ” usamos UUID,
   333вҶ’    -- asГӯ que comparamos por created_at).
   334вҶ’    DELETE FROM subscription_history a
   335вҶ’    USING subscription_history b
   336вҶ’    WHERE a.organization_id IS NOT DISTINCT FROM b.organization_id
   337вҶ’      AND a.event_type = b.event_type
   338вҶ’      AND a.details::text IS NOT DISTINCT FROM b.details::text
   339вҶ’      AND a.created_at < b.created_at;
   340вҶ’
   341вҶ’    -- 6b. Crear el Гӯndice UNIQUE
   342вҶ’    CREATE UNIQUE INDEX IF NOT EXISTS subscription_history_org_event_details_uniq
   343вҶ’      ON subscription_history(organization_id, event_type, details);
   344вҶ’  END IF;
   345вҶ’END $$;

   361вҶ’-- 7. ГҚNDICES FALTANTES EN FKs CRГҚTICAS
   362вҶ’-- ============================================================
   363вҶ’-- Cada CREATE INDEX se envuelve en un DO $$ que verifica si la
   364вҶ’-- tabla existe antes de intentar crear el Гӯndice. AsГӯ no falla
   365вҶ’-- si alguna migraciГіn intermedia no se ejecutГі.
   366вҶ’
   367вҶ’DO $$
   368вҶ’DECLARE
   369вҶ’  v_table text;
   370вҶ’  v_index text;
   371вҶ’  v_col text;
   372вҶ’BEGIN
   373вҶ’  -- (tabla, Гӯndice, columna)
   374вҶ’  FOR v_table, v_index, v_col IN SELECT * FROM (VALUES
   375вҶ’    ('order_items',         'order_items_menu_item_id_idx',     'menu_item_id'),
   376вҶ’    ('orders',              'orders_table_id_idx',              'table_id'),
   377вҶ’    ('reservations',        'reservations_table_id_idx',        'table_id'),
   378вҶ’    ('chat_messages',       'chat_messages_user_id_idx',        'user_id'),
   379вҶ’    ('role_permissions',    'role_permissions_permission_id_idx','permission_id'),
   380вҶ’    ('user_roles',          'user_roles_role_id_idx',           'role_id'),
   381вҶ’    ('order_items',         'order_items_order_id_idx',         'order_id'),
   382вҶ’    ('reservations',        'reservations_customer_id_idx',     'customer_id'),
   383вҶ’    ('menu_items',          'menu_items_category_id_idx',       'category_id'),
   384вҶ’    ('reservations',        'reservations_status_idx',          'status'),
   385вҶ’    ('reservations',        'reservations_date_idx',            'date'),
   386вҶ’    ('orders',              'orders_status_idx',                'status')
   387вҶ’  ) AS t(tab, idx, col)
   388вҶ’  LOOP
   389вҶ’    IF EXISTS (SELECT 1 FROM information_schema.columns
   390вҶ’               WHERE table_schema = 'public' AND table_name = v_table
   391вҶ’                 AND column_name = v_col) THEN
   392вҶ’      EXECUTE format('CREATE INDEX IF NOT EXISTS %I ON %I(%I);', v_index, v_table, v_col);
   393вҶ’    END IF;
   394вҶ’  END LOOP;
   395вҶ’
   396вҶ’  -- orders_created_at_idx usa ORDER BY (desc) que requiere sintaxis especial
   397вҶ’  IF EXISTS (SELECT 1 FROM information_schema.columns
   398вҶ’             WHERE table_schema = 'public' AND table_name = 'orders'
   399вҶ’               AND column_name = 'created_at') THEN
   400вҶ’    CREATE INDEX IF NOT EXISTS orders_created_at_idx ON orders(created_at DESC);
   401вҶ’  END IF;
   402вҶ’END $$;

   406вҶ’-- ============================================================
   407вҶ’DO $$
   408вҶ’DECLARE
   409вҶ’  t text;
   410вҶ’BEGIN
   411вҶ’  FOR t IN SELECT unnest(ARRAY[
   412вҶ’    'zones','customers','customer_tags','staff_shifts','chat_channels',
   413вҶ’    'subscription_plans','feature_flag_overrides','system_settings',
   414вҶ’    'email_queue','organization_usage','invoices','table_groups'
   415вҶ’  ])
   416вҶ’  LOOP
   417вҶ’    IF EXISTS (SELECT 1 FROM information_schema.columns
   418вҶ’               WHERE table_schema = 'public' AND table_name = t
   419вҶ’                 AND column_name = 'updated_at') THEN
   420вҶ’      EXECUTE format('DROP TRIGGER IF EXISTS %I_touch ON %I;', t, t);
   421вҶ’      EXECUTE format(
   422вҶ’        'CREATE TRIGGER %I_touch BEFORE UPDATE ON %I FOR EACH ROW EXECUTE FUNCTION touch_updated_at();',
   423вҶ’        t, t
   424вҶ’      );
   425вҶ’    END IF;
   426вҶ’  END LOOP;
   427вҶ’END $$;

   429вҶ’-- ============================================================
   430вҶ’-- 9. UNIQUE EN customers(organization_id, phone) y (organization_id, email)
   431вҶ’-- ============================================================
   432вҶ’-- IMPORTANTE: Si la DB ya contiene duplicados (mismo phone o email
   433вҶ’-- repetido en la misma organizaciГіn), el CREATE UNIQUE INDEX fallarГЎ
   434вҶ’-- con error 23505. Por eso primero de-duplicamos manteniendo solo
   435вҶ’-- la fila mГЎs reciente (MAX(created_at)) y reasignando cualquier
   436вҶ’-- referencia externa a la fila superviviente.
   437вҶ’
   438вҶ’-- 9a. De-duplicar por (organization_id, phone)
   439вҶ’-- Mantener la fila con el created_at mГЎs reciente; borrar el resto.
   440вҶ’-- Antes de borrar, mover cualquier reservation/order que apunte a
   441вҶ’-- los duplicados para que apunte al superviviente.
   442вҶ’DO $$
   443вҶ’DECLARE
   444вҶ’  dup_row record;
   445вҶ’  survivor_id uuid;
   446вҶ’BEGIN
   447вҶ’  FOR dup_row IN
   448вҶ’    SELECT organization_id, phone
   449вҶ’    FROM customers
   450вҶ’    WHERE phone IS NOT NULL
   451вҶ’    GROUP BY organization_id, phone
   452вҶ’    HAVING COUNT(*) > 1
   453вҶ’  LOOP
   454вҶ’    -- Elegir el superviviente: el de created_at mГЎs reciente
   455вҶ’    SELECT id INTO survivor_id
   456вҶ’    FROM customers
   457вҶ’    WHERE organization_id = dup_row.organization_id
   458вҶ’      AND phone = dup_row.phone
   459вҶ’    ORDER BY created_at DESC, updated_at DESC
   460вҶ’    LIMIT 1;
   461вҶ’
   462вҶ’    -- Mover reservations que apuntaban a duplicados
   463вҶ’    UPDATE reservations
   464вҶ’    SET customer_id = survivor_id
   465вҶ’    WHERE customer_id IN (
   466вҶ’      SELECT id FROM customers
   467вҶ’      WHERE organization_id = dup_row.organization_id
   468вҶ’        AND phone = dup_row.phone
   469вҶ’        AND id <> survivor_id
   470вҶ’    );
   471вҶ’
   472вҶ’    -- Mover cualquier otra referencia (orders, etc.) вҖ” usando
   473вҶ’    -- bloque anГіnimo para no fallar si la columna no existe.
   474вҶ’    BEGIN
   475вҶ’      UPDATE orders
   476вҶ’      SET customer_id = survivor_id
   477вҶ’      WHERE customer_id IN (
   478вҶ’        SELECT id FROM customers
   479вҶ’        WHERE organization_id = dup_row.organization_id
   480вҶ’          AND phone = dup_row.phone
   481вҶ’          AND id <> survivor_id
   482вҶ’      );
   483вҶ’    EXCEPTION WHEN OTHERS THEN NULL;
   484вҶ’    END;
   485вҶ’
   486вҶ’    -- Borrar los duplicados (todo menos el superviviente)
   487вҶ’    DELETE FROM customers
   488вҶ’    WHERE organization_id = dup_row.organization_id
   489вҶ’      AND phone = dup_row.phone
   490вҶ’      AND id <> survivor_id;
   491вҶ’  END LOOP;
   492вҶ’END $$;
   493вҶ’
   494вҶ’-- 9b. De-duplicar por (organization_id, email) вҖ” mismo patrГіn
   495вҶ’DO $$
   496вҶ’DECLARE
   497вҶ’  dup_row record;
   498вҶ’  survivor_id uuid;
   499вҶ’BEGIN
   500вҶ’  FOR dup_row IN
   501вҶ’    SELECT organization_id, email
   502вҶ’    FROM customers
   503вҶ’    WHERE email IS NOT NULL
   504вҶ’    GROUP BY organization_id, email
   505вҶ’    HAVING COUNT(*) > 1
   506вҶ’  LOOP
   507вҶ’    SELECT id INTO survivor_id
   508вҶ’    FROM customers
   509вҶ’    WHERE organization_id = dup_row.organization_id
   510вҶ’      AND email = dup_row.email
   511вҶ’    ORDER BY created_at DESC, updated_at DESC
   512вҶ’    LIMIT 1;
   513вҶ’
   514вҶ’    UPDATE reservations
   515вҶ’    SET customer_id = survivor_id
   516вҶ’    WHERE customer_id IN (
   517вҶ’      SELECT id FROM customers
   518вҶ’      WHERE organization_id = dup_row.organization_id
   519вҶ’        AND email = dup_row.email
   520вҶ’        AND id <> survivor_id
   521вҶ’    );
   522вҶ’
   523вҶ’    BEGIN
   524вҶ’      UPDATE orders
   525вҶ’      SET customer_id = survivor_id
   526вҶ’      WHERE customer_id IN (
   527вҶ’        SELECT id FROM customers
   528вҶ’        WHERE organization_id = dup_row.organization_id
   529вҶ’          AND email = dup_row.email
   530вҶ’          AND id <> survivor_id
   531вҶ’      );
   532вҶ’    EXCEPTION WHEN OTHERS THEN NULL;
   533вҶ’    END;
   534вҶ’
   535вҶ’    DELETE FROM customers
   536вҶ’    WHERE organization_id = dup_row.organization_id
   537вҶ’      AND email = dup_row.email
   538вҶ’      AND id <> survivor_id;
   539вҶ’  END LOOP;
   540вҶ’END $$;
